create a typescript project steps:
mkdir project-name
cd project-name
npm init
tsc --init
edit tsconfig.json to edit restrictions
git init
touch .gitignore
open browser go to url: gitignore.io
tsc *.ts -t ES2015 --experimentalDecorators
git status
git add .
git status
git commit -m 'initialize repo'
npm install typescript --save-dev
npm install ts-node --save-dev
